# models.py
from sqlalchemy import Column, String, Text, ForeignKey
from sqlalchemy.orm import relationship
from sqlalchemy.ext.declarative import declarative_base

Base = declarative_base()

class ParentDocument(Base):
    """
    【父表】：存放长文本片段或结构化章节摘要。不存储原始 PDF 字节流，只存路径及解析后的纯文本。
    """
    __tablename__ = 'parent_documents'
    
    id = Column(String(36), primary_key=True)
    file_name = Column(String(255), nullable=False)
    file_path = Column(String(512), nullable=False)        # 预留物理路径，用于前端追溯与下载
    full_parent_text = Column(Text, nullable=False)        # 2000字左右的清洗后文本，供大模型最终读取
    security_level = Column(String(20), default="public")  # RBAC 隔离字段

    children = relationship("ChildChunk", back_populates="parent", cascade="all, delete-orphan")

class ChildChunk(Base):
    """
    【子表】：存放细粒度短文本片段，与向量库的 Point ID 保持 1:1 特征映射
    """
    __tablename__ = 'child_chunks'
    
    id = Column(String(36), primary_key=True)
    parent_id = Column(String(36), ForeignKey('parent_documents.id', ondelete='CASCADE')) # 核心名词物理暗桩：parent_id
    child_text = Column(Text, nullable=False)              # 300字高聚焦纯文本（用于向量化输入）
    qdrant_point_id = Column(String(36))                   # 对应 Qdrant 里的 Point UUID
    
    parent = relationship("ParentDocument", back_populates="children")

# filter.py
async def execute_dynamic_window_filtering(query: str, raw_db_chunks: list, settings: RerankSettings):
    """
    基于 Cross-Encoder 绝对置信度分数的弹性滑窗截断逻辑
    """
    # 拼装交叉注意力的打分对 [[query, text1], [query, text2] ... ]
    input_pairs = [[query, chunk.child_text] for chunk in raw_db_chunks]
    ce_scores = await cross_encoder_service.predict(input_pairs) # 跨服务调用隔离的模型网关
    
    eligible_chunks = []
    for chunk, score in zip(raw_db_chunks, ce_scores):
        if score >= settings.score_threshold: # 依据三：基于 0.65 置信度硬红线过滤低相关噪音
            chunk.meta['score'] = score
            eligible_chunks.append(chunk)
            
    # 按得分重排，并根据显存上限约束进行弹性 K 值截断 (elastic_min_k=2, elastic_max_k=12)
    eligible_chunks = sorted(eligible_chunks, key=lambda x: x.meta['score'], reverse=True)
    return eligible_chunks[:settings.elastic_max_k]
